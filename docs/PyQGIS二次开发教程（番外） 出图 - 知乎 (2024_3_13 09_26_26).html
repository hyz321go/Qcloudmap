有人在评论区询问怎么出图，其实自动出图是一个比较麻烦的事，但是如果定义好了QPT模板，可以简化许多格式上的操作问题，下面是一个简单的示例：

# cython:language_level=3
# -*- coding: utf-8 -*-
from PyQt5.QtCore import Qt
from qgis.PyQt.QtXml import QDomDocument
from qgis.core import QgsProject,QgsLayout,QgsReadWriteContext,\
    QgsLayoutExporter,QgsPrintLayout,QgsApplication,QgsLayoutItemPage,\
    QgsRasterLayer,QgsLayoutItemMap,QgsLayoutItemLabel
import os.path as osp
from utils.yoyiDefault import yoyiDefaultValueClass
PROJECT = QgsProject.instance()

class yoyiPrintMapClass:
    def __init__(self,mapCanvas):
        resourceDir = yoyiDefaultValueClass().resourceDir
        self.qptDict = {
            "VerticalA": osp.join(resourceDir,"verticalA.qpt"),
            "HorizontalB" : osp.join(resourceDir,"horizontalB.qpt")
        }
        self.mapCanvas = mapCanvas
    # A 竖版简单出图
    # 只有渲染框、标题、指北针、标尺、简单图例
    # id说明：
    # 1 标题
    # 2 渲染框
    # 3 简单图例 可以更改简单图例的颜色
    def printMapVerticalA(self,titleText,outPath):
        with open(self.qptDict["VerticalA"]) as f:
            template_content = f.read()
        doc = QDomDocument()
        doc.setContent(template_content)
        layout = QgsLayout(PROJECT)
        layout.initializeDefaults()
        # 将横板调为竖版
        layout.pageCollection().page(0).setPageSize('A4', QgsLayoutItemPage.Orientation.Portrait)
        # 开始读取模板文件
        tems, ok = layout.loadFromTemplate(doc, QgsReadWriteContext(), False)
        # 将现在的地图画布渲染到框中
        mapItem : QgsLayoutItemMap = layout.itemById("2")
        mapItem.zoomToExtent(self.mapCanvas.extent())
        # 修改标题的文字
        titleItem : QgsLayoutItemLabel = layout.itemById("1")
        titleItem.setText(titleText)
        # 修改
        # 保存到文件中
        exporter = QgsLayoutExporter(layout)
        exporter.exportToImage(outPath,QgsLayoutExporter.ImageExportSettings())


主要关注 QgsLayout QgsReadWriteContext QgsLayoutItemLabel QgsLayoutExporter

以及将mapCanvas 的内容渲染到出图中。
